```rust
// framework/src/schema/mod.rs
//! Thin facade over sea-query schema + query builders. The full
//! sea-query API is reachable via the underlying types we re-export;
//! the facade is Laravel-shape ergonomics on top.

pub use sea_query::*; // expose ColumnDef, Table, Index, Query, etc.

/// Helper for `Schema::create_table("users", |t| { ... })`.
pub struct Schema;

impl Schema {
    pub fn create_table<F>(name: &str, build: F) -> sea_query::TableCreateStatement
    where
        F: FnOnce(&mut TableBuilder),
    {
        let mut builder = TableBuilder::new(name);
        build(&mut builder);
        builder.into_statement()
    }

    pub fn alter_table<F>(name: &str, build: F) -> sea_query::TableAlterStatement
    where
        F: FnOnce(&mut sea_query::TableAlterStatement),
    {
        let mut stmt = sea_query::Table::alter().table(sea_query::Alias::new(name)).to_owned();
        build(&mut stmt);
        stmt
    }

    pub fn drop_table(name: &str) -> sea_query::TableDropStatement {
        sea_query::Table::drop().table(sea_query::Alias::new(name)).to_owned()
    }
}

pub struct TableBuilder {
    name: String,
    columns: Vec<sea_query::ColumnDef>,
}

impl TableBuilder {
    fn new(name: &str) -> Self {
        Self {
            name: name.to_string(),
            columns: Vec::new(),
        }
    }

    pub fn column(&mut self, def: sea_query::ColumnDef) -> &mut Self {
        self.columns.push(def);
        self
    }

    fn into_statement(self) -> sea_query::TableCreateStatement {
        let mut stmt = sea_query::Table::create()
            .table(sea_query::Alias::new(&self.name))
            .if_not_exists()
            .to_owned();
        for col in self.columns {
            stmt.col(col);
        }
        stmt
    }
}

pub mod column_helpers {
    use sea_query::{Alias, ColumnDef, ColumnType};

    pub fn big_integer(name: &str) -> ColumnDef {
        ColumnDef::new_with_type(Alias::new(name), ColumnType::BigInteger)
    }
    pub fn string(name: &str) -> ColumnDef {
        ColumnDef::new_with_type(Alias::new(name), ColumnType::String(Some(255)))
    }
    pub fn datetime(name: &str) -> ColumnDef {
        ColumnDef::new_with_type(Alias::new(name), ColumnType::DateTime)
    }
    pub fn text(name: &str) -> ColumnDef {
        ColumnDef::new_with_type(Alias::new(name), ColumnType::Text)
    }
}
```

```rust
/// Parse an English schedule string (e.g. "every 5 minutes",
/// "every day at 8am", "every monday at noon") into a cron

pub fn english_to_cron(input: &str) -> Result<CronExpression, FrameworkError> {

        .map_err(|e| FrameworkError::internal(format!("english-to-cron: {}", e)))?;
```

```rust
// framework/src/schedule/mod.rs — extend TaskBuilder::at
impl TaskBuilder {
    /// Schedule using either a raw cron expression (`"0 */5 * * *"`)
    /// or a natural-language string (`"every 5 minutes"`).
    pub fn at(mut self, schedule: &str) -> Self {
        let cron = if schedule.contains(' ') && !schedule.starts_with(|c: char| c.is_ascii_digit() || c == '*') {
            // Probably English (contains spaces and no leading digit/asterisk)
            crate::schedule::english::english_to_cron(schedule)
                .expect("invalid schedule string")
        } else {
            CronExpression::parse(schedule).expect("invalid cron")
        };
        self.cron = cron;
        self
    }
}
```

```rust
// framework/src/i18n/mod.rs
//! Simple JSON-based translations. Per-locale files in
//! `lang/<locale>.json`. Nested keys with `.` separator; placeholder
//! substitution with `:name`.

mod loader;

use std::collections::HashMap;
use std::sync::{Mutex, OnceLock};

static CURRENT: OnceLock<Mutex<String>> = OnceLock::new();
static REGISTRY: OnceLock<Mutex<HashMap<String, serde_json::Value>>> = OnceLock::new();

fn current() -> &'static Mutex<String> {
    CURRENT.get_or_init(|| Mutex::new("en".into()))
}

fn registry() -> &'static Mutex<HashMap<String, serde_json::Value>> {
    REGISTRY.get_or_init(|| Mutex::new(HashMap::new()))
}

pub struct Lang;

impl Lang {
    pub fn set(locale: impl Into<String>) {
        *current().lock().unwrap() = locale.into();
    }
    pub fn current_locale() -> String {
        current().lock().unwrap().clone()
    }
    pub fn register_inline(locale: impl Into<String>, json: serde_json::Value) {
        registry().lock().unwrap().insert(locale.into(), json);
    }
    pub fn load_from_dir(dir: impl AsRef<std::path::Path>) -> Result<(), crate::FrameworkError> {
        loader::load(dir.as_ref())
    }
}

/// Translate a key, optionally substituting `:placeholders`.
pub fn t(key: &str, replacements: &[(&str, &str)]) -> String {
    let locale = Lang::current_locale();
    let reg = registry().lock().unwrap();

    let lookup = |loc: &str| -> Option<String> {
        let value = reg.get(loc)?;
        let mut current = value;
        for segment in key.split('.') {
            current = current.get(segment)?;
        }
        current.as_str().map(|s| s.to_string())
    };

    let template = lookup(&locale)
        .or_else(|| lookup("en"))
        .unwrap_or_else(|| key.to_string());

    let mut out = template;
    for (k, v) in replacements {
        out = out.replace(&format!(":{}", k), v);
    }
    out
}
```

```rust
use crate::FrameworkError;
use std::path::Path;

pub fn load(dir: &Path) -> Result<(), FrameworkError> {
    for entry in std::fs::read_dir(dir)
        .map_err(|e| FrameworkError::internal(format!("read lang dir: {}", e)))?
    {
        let entry = entry.map_err(|e| FrameworkError::internal(format!("entry: {}", e)))?;
        if entry.path().extension().and_then(|s| s.to_str()) != Some("json") {
            continue;
        }
        let locale = entry.path().file_stem().unwrap().to_string_lossy().into_owned();
        let raw = std::fs::read_to_string(entry.path())
            .map_err(|e| FrameworkError::internal(format!("read {}: {}", entry.path().display(), e)))?;
        let json: serde_json::Value = serde_json::from_str(&raw)
            .map_err(|e| FrameworkError::internal(format!("parse: {}", e)))?;
        super::Lang::register_inline(locale, json);
    }
    Ok(())
}
```

```rust
// framework/src/support/string.rs
use convert_case::{Case, Casing};

pub struct Str;

impl Str {
    pub fn camel(s: &str) -> String {
        s.to_case(Case::Camel)
    }
    pub fn snake(s: &str) -> String {
        s.to_case(Case::Snake)
    }
    pub fn studly(s: &str) -> String {
        s.to_case(Case::Pascal)
    }
    pub fn slug(s: &str) -> String {
        let lower = s.to_lowercase();
        let mut out = String::with_capacity(lower.len());
        let mut last_dash = true;
        for c in lower.chars() {
            if c.is_ascii_alphanumeric() {
                out.push(c);
                last_dash = false;
            } else if !last_dash {
                out.push('-');
                last_dash = true;
            }
        }
        out.trim_matches('-').to_string()
    }
}
```

```rust
// framework/src/support/array.rs
use std::collections::HashMap;

pub struct Arr;

impl Arr {
    pub fn pluck<T, U: Clone, F: Fn(&T) -> U>(items: &[T], extract: F) -> Vec<U> {
        items.iter().map(|t| extract(t)).collect()
    }

    pub fn group_by<T: Clone, K: Eq + std::hash::Hash, F: Fn(&T) -> K>(
        items: &[T],
        key: F,
    ) -> HashMap<K, Vec<T>> {
        let mut map: HashMap<K, Vec<T>> = HashMap::new();
        for item in items {
            map.entry(key(item)).or_default().push(item.clone());
        }
        map
    }

    pub fn chunk<T: Clone>(items: &[T], size: usize) -> Vec<Vec<T>> {
        items.chunks(size).map(|c| c.to_vec()).collect()
    }
}
```

```rust
/// Walk a dotted path through a `serde_json::Value` tree.
pub fn data_get<'a>(value: &'a Value, path: &str) -> Option<&'a Value> {
    let mut current = value;
    for segment in path.split('.') {
        match current {
            Value::Object(map) => current = map.get(segment)?,
            Value::Array(arr) => {
                let idx: usize = segment.parse().ok()?;
                current = arr.get(idx)?;
            }
            _ => return None,
        }
    }
    Some(current)
}

/// True if the value is blank (empty string / whitespace).
pub fn blank<S: AsRef<str>>(s: &S) -> bool {
    s.as_ref().trim().is_empty()
}

/// True if the value is non-blank.
pub fn filled<S: AsRef<str>>(s: &S) -> bool {

}

/// Run `f`; on `Err`, return `default`. Laravel's `rescue($try, $rescue)`.
pub async fn rescue<F, Fut, T, E>(f: F, default: T) -> T
where
    F: FnOnce() -> Fut,
    Fut: Future<Output = Result<T, E>>,
{
    f().await.unwrap_or(default)
}

/// Try `f` up to `attempts` times, sleeping `delay` between failures.
pub async fn retry<F, Fut, T, E>(attempts: u32, delay: std::time::Duration, mut f: F) -> Result<T, E>
where
    F: FnMut() -> Fut,
    Fut: Future<Output = Result<T, E>>,
{
    let mut last_err = None;
    for _ in 0..attempts {
        match f().await {
            Ok(v) => return Ok(v),
            Err(e) => {
                last_err = Some(e);
                tokio::time::sleep(delay).await;
            }
        }
    }
    Err(last_err.expect("attempts > 0"))
}
```

```rust
pub struct Process;

pub struct ProcessBuilder {
    command: String,
    cwd: Option<std::path::PathBuf>,
    envs: Vec<(String, String)>,
}

impl ProcessBuilder {
    pub fn cwd(mut self, path: impl Into<std::path::PathBuf>) -> Self {
        self.cwd = Some(path.into());
        self
    }
    pub fn env(mut self, k: impl Into<String>, v: impl Into<String>) -> Self {
        self.envs.push((k.into(), v.into()));
        self
    }
    pub async fn output(self) -> Result<ProcessOutput, FrameworkError> {
        let parts: Vec<&str> = self.command.split_whitespace().collect();
        let mut cmd = Command::new(parts[0]);
        cmd.args(&parts[1..]);
        if let Some(cwd) = &self.cwd {
            cmd.current_dir(cwd);
        }
        for (k, v) in &self.envs {
            cmd.env(k, v);
        }
        let out = cmd
            .output()
            .await
            .map_err(|e| FrameworkError::internal(format!("process: {}", e)))?;
        Ok(ProcessOutput {
            status: out.status.code().unwrap_or(-1),
            stdout: String::from_utf8_lossy(&out.stdout).into_owned(),
            stderr: String::from_utf8_lossy(&out.stderr).into_owned(),
        })
    }
}

pub struct ProcessOutput {
    pub status: i32,
    pub stdout: String,
    pub stderr: String,
}

    }
```

```rust
// framework/src/routing/signed.rs
use crate::{Encrypter, FrameworkError};
use std::time::{SystemTime, UNIX_EPOCH};

pub fn signed_url(
    base: &str,
    params: &[(&str, &str)],
    expires_in: std::time::Duration,
    encrypter: &Encrypter,
) -> Result<String, FrameworkError> {
    let expires = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs() + expires_in.as_secs();
    let mut query: Vec<(String, String)> = params.iter().map(|(k, v)| (k.to_string(), v.to_string())).collect();
    query.push(("expires".into(), expires.to_string()));
    let canonical = serde_urlencoded::to_string(&query)
        .map_err(|e| FrameworkError::internal(format!("encode: {}", e)))?;
    let signature = encrypter.encrypt_string(&canonical)?;
    Ok(format!("{}?{}&sig={}", base, canonical, signature))
}

pub fn verify(url: &str, encrypter: &Encrypter) -> Result<bool, FrameworkError> {
    let (path, query) = url.split_once('?').ok_or_else(|| FrameworkError::param("query"))?;
    let mut pairs: Vec<(String, String)> = serde_urlencoded::from_str(query)
        .map_err(|e| FrameworkError::internal(format!("decode: {}", e)))?;
    let sig = pairs
        .iter()
        .position(|(k, _)| k == "sig")
        .map(|i| pairs.remove(i).1)
        .ok_or_else(|| FrameworkError::param("sig"))?;
    let canonical = serde_urlencoded::to_string(&pairs)
        .map_err(|e| FrameworkError::internal(format!("encode: {}", e)))?;
    let decrypted = encrypter.decrypt_string(&sig)?;
    if decrypted != canonical {
        return Ok(false);
    }
    let expires: u64 = pairs
        .iter()
        .find(|(k, _)| k == "expires")
        .and_then(|(_, v)| v.parse().ok())
        .unwrap_or(0);
    let now = SystemTime::now().duration_since(UNIX_EPOCH).unwrap().as_secs();
    let _ = path;
    Ok(now < expires)
}
```

```rust
// framework/src/routing/reverse.rs
//! `route("users.show", &[("id", "42")])` → "/users/42"
//!
//! The named-route registry is populated by the `routes!` macro at
//! compile time via inventory.

use crate::FrameworkError;
use std::collections::HashMap;
use std::sync::Mutex;

static REGISTRY: Mutex<Option<HashMap<String, String>>> = Mutex::new(None);

pub fn register(name: impl Into<String>, pattern: impl Into<String>) {
    let mut g = REGISTRY.lock().unwrap();
    let map = g.get_or_insert_with(HashMap::new);
    map.insert(name.into(), pattern.into());
}

pub fn url(name: &str, params: &[(&str, &str)]) -> Result<String, FrameworkError> {
    let pattern = {
        let g = REGISTRY.lock().unwrap();
        g.as_ref()
            .and_then(|m| m.get(name).cloned())
            .ok_or_else(|| FrameworkError::internal(format!("unknown route name: {}", name)))?
    };
    let mut out = pattern;
    for (k, v) in params {
        out = out.replace(&format!(":{}", k), v);
    }
    Ok(out)
}
```

```rust
// framework/src/container/scoped.rs
//! Per-request scoped container built on `dependency-injector`.
//!
//! The process holds one root `Container` (process-wide singletons).
//! Each request installs a child container via `Container::scope()`
//! held in a `tokio::task_local!` — so request-scoped resolves see
//! request-specific overrides, falling through to root for everything
//! else.

use std::sync::Arc;

// Process-wide root container. Constructed once on first access.
static ROOT: std::sync::OnceLock<Arc<dependency_injector::Container>> = std::sync::OnceLock::new();

tokio::task_local! {
    /// The current request's scoped child container. None outside a
    /// request scope (background jobs, tests, etc.) — resolves fall
    /// through to root.
    pub(crate) static REQUEST_SCOPE: Arc<dependency_injector::Container>;
}

pub struct Scoped;

impl Scoped {
    /// Returns the process-wide root container. Use this to register
    /// process-singleton services at boot.
    pub fn root() -> &'static dependency_injector::Container {
        ROOT.get_or_init(|| Arc::new(dependency_injector::Container::new()))
    }

    /// Create a new child container scoped to the next request. The
    /// child inherits all bindings from `root` and can override them.
    pub async fn new_child() -> Arc<dependency_injector::Container> {
        Arc::new(Self::root().scope())
    }

    /// Resolve a service in the current request's scope, falling back
    /// to root. Outside a request scope, resolves only from root.
    pub fn resolve<T: Send + Sync + 'static>() -> Option<Arc<T>> {
        REQUEST_SCOPE
            .try_with(|child| child.resolve::<T>())
            .ok()
            .flatten()
            .or_else(|| Self::root().resolve::<T>())
    }
}
```

```rust
use sea_orm::{ConnectionTrait, DatabaseTransaction, Statement, TransactionTrait};
use serde::de::DeserializeOwned;
use std::future::Future;

impl DB {
    /// Run a closure inside a transaction. Commits on `Ok`, rolls back on `Err`.
    pub async fn transaction<F, Fut, T, E>(f: F) -> Result<T, E>
    where
        F: for<'c> FnOnce(&'c DatabaseTransaction) -> Fut + Send,
        Fut: Future<Output = Result<T, E>> + Send,
        E: From<FrameworkError> + Send + 'static,
        T: Send + 'static,
    {
        let conn = Self::connection().await?;
        conn.transaction::<_, T, E>(|tx| Box::pin(async move { f(tx).await }))
            .await
            .map_err(|err| match err {
                sea_orm::TransactionError::Connection(e) => {
                    FrameworkError::internal(format!("db transaction: {}", e)).into()
                }
                sea_orm::TransactionError::Transaction(inner) => inner,
            })
    }

    /// Same as `transaction` but retry on transient serialization
    /// failures (PG deadlock, MySQL deadlock) up to `attempts`.
    pub async fn transaction_with_retry<F, Fut, T, E>(attempts: u32, mut f: F) -> Result<T, E>
    where
        F: for<'c> FnMut(&'c DatabaseTransaction) -> Fut + Send,
        Fut: Future<Output = Result<T, E>> + Send,
        E: From<FrameworkError> + std::fmt::Display + Send + 'static,
        T: Send + 'static,
    {
        let mut last_err: Option<E> = None;
        for attempt in 1..=attempts {
            match Self::transaction(&mut f).await {
                Ok(v) => return Ok(v),
                Err(e) => {
                    let msg = format!("{}", e);
                    if !msg.contains("deadlock") && !msg.contains("serialization") {
                        return Err(e);
                    }
                    tracing::warn!(attempt, error = %e, "db transaction retry");
                    last_err = Some(e);
                }
            }
        }
        Err(last_err.unwrap())
    }

    /// Open a named savepoint inside an active transaction. The `f`
    /// closure runs against the same transaction; on `Err`, only the
    /// savepoint's changes are rolled back.
    pub async fn savepoint<F, Fut, T, E>(
        tx: &DatabaseTransaction,
        name: &str,
        f: F,
    ) -> Result<T, E>
    where
        F: for<'c> FnOnce(&'c DatabaseTransaction) -> Fut + Send,
        Fut: Future<Output = Result<T, E>> + Send,
        E: From<FrameworkError> + Send + 'static,
    {
        let safe_name = name
            .chars()
            .filter(|c| c.is_ascii_alphanumeric() || *c == '_')
            .collect::<String>();
        tx.execute_unprepared(&format!("SAVEPOINT {}", safe_name))
            .await
            .map_err(|e| FrameworkError::internal(format!("savepoint: {}", e)).into())?;
        match f(tx).await {
            Ok(v) => {
                tx.execute_unprepared(&format!("RELEASE SAVEPOINT {}", safe_name))
                    .await
                    .map_err(|e| FrameworkError::internal(format!("release sp: {}", e)).into())?;
                Ok(v)
            }
            Err(err) => {
                tx.execute_unprepared(&format!("ROLLBACK TO SAVEPOINT {}", safe_name))
                    .await
                    .map_err(|e| FrameworkError::internal(format!("rollback sp: {}", e)))?;
                Err(err)
            }
        }
    }

    /// Run an ad-hoc SELECT and deserialize each row into `T`.
    pub async fn raw<T: DeserializeOwned + Send + 'static>(
        sql: &str,
        bindings: &[sea_orm::Value],
    ) -> Result<Vec<T>, FrameworkError> {
        let conn = Self::connection().await?;
        let stmt = Statement::from_sql_and_values(conn.get_database_backend(), sql, bindings.to_vec());
        let rows = conn
            .query_all(stmt)
            .await
            .map_err(|e| FrameworkError::internal(format!("db raw: {}", e)))?;
        rows.into_iter()
            .map(|r| {
                let json = sea_orm_row_to_json(&r);
                serde_json::from_value(json)
                    .map_err(|e| FrameworkError::internal(format!("db raw decode: {}", e)))
            })
            .collect()
    }

    /// Run an ad-hoc DML statement (INSERT/UPDATE/DELETE).
    pub async fn statement(
        sql: &str,
        bindings: &[sea_orm::Value],
    ) -> Result<u64, FrameworkError> {
        let conn = Self::connection().await?;
        let stmt = Statement::from_sql_and_values(conn.get_database_backend(), sql, bindings.to_vec());
        let result = conn
            .execute(stmt)
            .await
            .map_err(|e| FrameworkError::internal(format!("db statement: {}", e)))?;
        Ok(result.rows_affected())
    }

    /// Variant of `statement` that runs inside an active transaction.
    pub async fn statement_in(
        tx: &DatabaseTransaction,
        sql: &str,
        bindings: &[sea_orm::Value],
    ) -> Result<u64, FrameworkError> {
        let stmt = Statement::from_sql_and_values(tx.get_database_backend(), sql, bindings.to_vec());
        let result = tx
            .execute(stmt)
            .await
            .map_err(|e| FrameworkError::internal(format!("db statement_in: {}", e)))?;

    }
}
```

```rust
// framework/src/pipeline/mod.rs
//! Generic value-through-handlers pipeline.

use futures::future::BoxFuture;
use std::future::Future;
use std::sync::Arc;

type HandlerFn<T> = Arc<
    dyn for<'a> Fn(T, Next<T>) -> BoxFuture<'a, T> + Send + Sync,
>;

pub struct Pipeline<T> {
    value: T,
    handlers: Vec<HandlerFn<T>>,
}

pub struct Next<T> {
    handlers: Vec<HandlerFn<T>>,
}

impl<T: Send + Sync + 'static> Next<T> {
    pub async fn run(self, value: T) -> T {
        let mut handlers = self.handlers;
        if let Some(h) = handlers.pop() {
            let inner = Next { handlers };
            h(value, inner).await
        } else {
            value
        }
    }
}

impl<T: Send + Sync + 'static> Pipeline<T> {
    pub fn send(value: T) -> Self {
        Self { value, handlers: Vec::new() }
    }

    pub fn through<F>(mut self, f: F) -> Self
    where
        F: for<'a> Fn(T, Next<T>) -> BoxFuture<'a, T> + Send + Sync + 'static,
    {
        self.handlers.insert(0, Arc::new(f));
        self
    }

    pub async fn then_return(self) -> T {
        let next = Next { handlers: self.handlers };
        next.run(self.value).await
    }

    pub async fn then<R, F, Fut>(self, terminal: F) -> R
    where
        F: FnOnce(T) -> Fut,
        Fut: Future<Output = R>,
    {
        let final_value = self.then_return().await;
        terminal(final_value).await
    }
}
```

```rust
// framework/src/csrf/middleware.rs — extend
use crate::http::{Request, Response};
use async_trait::async_trait;

#[derive(Debug, Clone, Default)]
pub struct CsrfConfig {
    pub allowed_origins: Vec<String>,
    pub except_paths: Vec<String>,
}

pub struct CsrfMiddleware {
    config: CsrfConfig,
}

impl CsrfMiddleware {
    pub fn new(config: CsrfConfig) -> Self {
        Self { config }
    }

    fn origin_allowed(&self, origin: Option<&str>) -> bool {
        match origin {
            None => true, // browsers omit Origin on same-origin top-level navs
            Some(o) => self.config.allowed_origins.iter().any(|allowed| allowed == o),
        }
    }
}

#[async_trait]
impl crate::middleware::Middleware for CsrfMiddleware {
    async fn handle(&self, request: Request, next: crate::middleware::Next) -> Response {
        // Skip non-mutating verbs
        let method = request.method();
        if matches!(method, "GET" | "HEAD" | "OPTIONS") {
            return next(request).await;
        }

        // Origin check (L13 PreventRequestForgery)
        let origin = request.header("origin");
        if !self.origin_allowed(origin.as_deref()) {
            return Err(crate::http::HttpResponse::status(419)
                .body("CSRF: origin mismatch"));
        }

        // Token check (existing behavior)
        let token = request.header("x-csrf-token")
            .or_else(|| request.form_field("_token"));
        let expected = crate::csrf::current_token();
        if token.as_deref() != Some(expected.as_str()) {
            return Err(crate::http::HttpResponse::status(419)
                .body("CSRF: token mismatch"));
        }

        next(request).await
    }
}
```

```rust
// framework/src/redis/mod.rs
//! `Redis::*` raw-command facade over the underlying `redis` crate.

use crate::FrameworkError;
use redis::AsyncCommands;
use redis::aio::ConnectionManager;
use std::sync::OnceLock;

static CONN: OnceLock<ConnectionManager> = OnceLock::new();

pub struct Redis;

impl Redis {
    pub(crate) async fn init(url: &str) -> Result<(), FrameworkError> {
        let client = redis::Client::open(url)
            .map_err(|e| FrameworkError::internal(format!("redis client: {}", e)))?;
        let mgr = client
            .get_connection_manager()
            .await
            .map_err(|e| FrameworkError::internal(format!("redis connect: {}", e)))?;
        let _ = CONN.set(mgr);
        Ok(())
    }

    fn conn() -> Result<ConnectionManager, FrameworkError> {
        CONN.get()
            .cloned()
            .ok_or_else(|| FrameworkError::internal("redis not initialized"))
    }

    pub async fn set(key: &str, value: &str) -> Result<(), FrameworkError> {
        let mut c = Self::conn()?;
        c.set::<_, _, ()>(key, value)
            .await
            .map_err(|e| FrameworkError::internal(format!("redis set: {}", e)))
    }

    pub async fn get(key: &str) -> Result<Option<String>, FrameworkError> {
        let mut c = Self::conn()?;
        c.get(key)
            .await
            .map_err(|e| FrameworkError::internal(format!("redis get: {}", e)))
    }

    pub async fn publish(channel: &str, message: &str) -> Result<(), FrameworkError> {
        let mut c = Self::conn()?;
        c.publish::<_, _, ()>(channel, message)
            .await
            .map_err(|e| FrameworkError::internal(format!("redis publish: {}", e)))
    }

    pub async fn subscribe(channel: &str) -> Result<Subscription, FrameworkError> {
        let client = redis::Client::open(crate::config::env_required("REDIS_URL"))
            .map_err(|e| FrameworkError::internal(format!("redis sub client: {}", e)))?;
        let mut pubsub = client
            .get_async_pubsub()
            .await
            .map_err(|e| FrameworkError::internal(format!("redis pubsub: {}", e)))?;
        pubsub
            .subscribe(channel)
            .await
            .map_err(|e| FrameworkError::internal(format!("redis subscribe: {}", e)))?;
        Ok(Subscription { inner: pubsub })
    }

    pub async fn script<T>(script: &str, keys: &[&str], args: &[&str]) -> Result<T, FrameworkError>
    where
        T: redis::FromRedisValue,
    {
        let mut c = Self::conn()?;
        let mut s = redis::cmd("EVAL");
        s.arg(script).arg(keys.len());
        for k in keys { s.arg(k); }
        for a in args { s.arg(a); }
        s.query_async(&mut c)
            .await
            .map_err(|e| FrameworkError::internal(format!("redis script: {}", e)))
    }

    /// Borrow the underlying connection for transactions, pipelines,
    /// or other multi-command flows.
    pub fn raw() -> Result<ConnectionManager, FrameworkError> {
        Self::conn()
    }
}

pub struct Subscription {
    inner: redis::aio::PubSub,
}

impl Subscription {
    pub async fn next_message(&mut self) -> Result<String, FrameworkError> {
        use futures::StreamExt;
        let mut stream = self.inner.on_message();
        let msg = stream
            .next()
            .await
            .ok_or_else(|| FrameworkError::internal("pubsub stream closed"))?;
        msg.get_payload()
            .map_err(|e| FrameworkError::internal(format!("redis msg decode: {}", e)))
    }
}
```

closure execution. The native Rust idiom is `tokio::try_join!` or
`JoinSet` — we wrap them under the Laravel-shaped facade for
migration ease.

**Files:** `framework/src/concurrency/mod.rs`

```rust
// framework/tests/concurrency.rs
use suprnova::Concurrency;

#[tokio::test]
async fn concurrency_run_executes_futures_in_parallel() {
    let started = std::time::Instant::now();
    let results: Vec<i32> = Concurrency::run(vec![
        Box::pin(async { tokio::time::sleep(std::time::Duration::from_millis(100)).await; 1 }),
        Box::pin(async { tokio::time::sleep(std::time::Duration::from_millis(100)).await; 2 }),
        Box::pin(async { tokio::time::sleep(std::time::Duration::from_millis(100)).await; 3 }),
    ]).await;
    let elapsed = started.elapsed();
    assert_eq!(results, vec![1, 2, 3]);
    assert!(elapsed < std::time::Duration::from_millis(250),
        "must run in parallel, elapsed={:?}", elapsed);
}
```

```rust
// framework/src/concurrency/mod.rs
//! `Concurrency::run([futures])` — Laravel-shaped wrapper over
//! `futures::future::join_all`. The Rust idiom is `try_join!` /
//! `JoinSet`; we offer this for migration ergonomics.

use futures::future::BoxFuture;

pub struct Concurrency;

impl Concurrency {
    /// Run all futures concurrently; return their results in order.
    pub async fn run<T>(futures: Vec<BoxFuture<'static, T>>) -> Vec<T> {
        futures::future::join_all(futures).await
    }

    /// Like `run` but each future returns `Result`. Returns `Err` on
    /// the first failure (matching `try_join_all` semantics).
    pub async fn try_run<T, E>(
        futures: Vec<BoxFuture<'static, Result<T, E>>>,
    ) -> Result<Vec<T>, E> {
        futures::future::try_join_all(futures).await
    }
}
```
